194<script setup>
195import { ref, watch, onMounted, onUnmounted } from 'vue'
196import { usePrinterStore } from '../stores/printer'
197import { deteccionErrores } from '../services/inferenceService'
198import { useToast } from 'primevue/usetoast'
199
200const store = usePrinterStore()
201const toast = useToast()
202
203// Estado de los paneles
204const configColapsado = ref(false)
205const monitorColapsado = ref(true)
206
207// Abrir el panel de monitorización automáticamente al empezar a imprimir
208watch(() => store.estaImprimiendo, (imprimiendo) => {
209 if (imprimiendo) {
210 configColapsado.value = true
211 monitorColapsado.value = false
212 }
213})
214
215const direccionIp = ref('')
216const archivoSeleccionado = ref('')
217
218// Referencias al elemento de la cámara y al canvas de captura
219const imagenCamara = ref(null)
220const canvasCaptura = ref(null)
221const camaraDisponible = ref(true)
222let intervaloInferencia = null
223let procesando = false
224
225onMounted(() => {
226 direccionIp.value = store.backendUrl || localStorage.getItem('printer_monitor_backend_url') || ''
227
228 // Cada 2s capturamos un frame de la cámara y lo pasamos al modelo para detectar errores
229 intervaloInferencia = setInterval(async () => {
230 if (!imagenCamara.value || !canvasCaptura.value || !store.urlStream || !camaraDisponible.value || procesando) return
231 if (!store.estaImprimiendo) return
232
233 procesando = true
234 try {
235 const imagen = imagenCamara.value
236 const canvas = canvasCaptura.value
237
238 // Ajustamos el canvas al tamaño del frame y lo dibujamos
239 canvas.width = imagen.naturalWidth || 640
240 canvas.height = imagen.naturalHeight || 480
241 if (canvas.width === 0 || canvas.height === 0) return
242
243 // El contexto 2D es el "pincel" que nos permite dibujar sobre el canvas
244 const ctx = canvas.getContext('2d')
245 // Copiamos el frame actual de la cámara al canvas
246 ctx.drawImage(imagen, 0, 0, canvas.width, canvas.height)
247
248 // Convertimos el canvas a un string base64 para poder enviárselo al modelo de IA
249 const capturaDataUrl = canvas.toDataURL('image/jpeg', 0.8)
250
251 // Enviamos el frame al modelo de IA y obtenemos si ha detectado algún error
252 const resultado = await deteccionErrores(capturaDataUrl)
253
254 // Guardamos el resultado en el store para que la UI pueda reaccionar
255 store.detecciones.has_errors = resultado.has_errors
256
257 // Si la IA detecta un error, registrarlo y cancelar la impresión
258 if (resultado.has_errors) {
259 console.warn('[IA] Error detectado, cancelando impresión...')
260 await store.registrarError()
261 store.enviarComando('cancel')
262 store.mensajeCancelacionAuto = 'Impresión cancelada automáticamente por detección de errores (Inferencia Local).'
263 }
264 } catch (err) {
265 console.error('[IA] Error capturando frame:', err)
266 } finally {
267 procesando = false
268 }
269 }, 2000)
270})
271
272// Limpiar el intervalo de inferencia al desmontar el componente
273onUnmounted(() => {
274 if (intervaloInferencia) clearInterval(intervaloInferencia)
275})
276
277// Manejar error de carga de la imagen de la cámara
278function alErrorImagen() {
279 camaraDisponible.value = false
280}
281
282// Cuando cambia la URL del stream (reconexión), volvemos a intentar mostrar la cámara
283watch(() => store.urlStream, (nuevaUrl) => {
284 if (nuevaUrl) {
285 camaraDisponible.value = true
286 }
287})
288
289// Sincronizar la IP del input con la del store
290watch(() => store.backendUrl, (nuevaUrl) => {
291 if (nuevaUrl) direccionIp.value = nuevaUrl
292})
293
294// Sincronizar el archivo seleccionado con el que tiene OctoPrint
295watch(() => store.estado.job?.file, (archivo) => {
296 if (archivo && archivo !== 'Sin archivo') {
297 archivoSeleccionado.value = archivo
298 }
299}, { immediate: true })
300
301// Mostrar un toast cuando se cancela automáticamente por error
302watch(() => store.mensajeCancelacionAuto, (mensaje) => {
303 if (mensaje) {
304 toast.add({ severity: 'error', summary: 'Impresión cancelada', detail: mensaje, life: 5000 })
305 store.limpiarMensajeCancelacion()
306 }
307})
308
309// Mostrar un toast cuando falla la conexión con el backend
310watch(() => store.errorConexion, (mensaje) => {
311 if (mensaje) {
312 toast.add({ severity: 'error', summary: 'Error de conexión', detail: mensaje, life: 5000 })
313 store.errorConexion = ''
314 }
315})
316
317// Actualizar la IP del backend y reconectar
318async function actualizarIp() {
319 let url = direccionIp.value.trim()
320 if (!url) return
321 url = url.replace(/^https?:\/\//, '')
322 store.guardarConfiguracion(url)
323 await store.conectar(true)
324}
325
326// Alternar entre conectar y desconectar
327async function alternarConexion() {
328 if (store.wsConectado) {
329 store.desconectarWebSocket(true)
330 toast.add({ severity: 'info', summary: 'Desconectado', detail: 'Conexión cerrada', life: 3000 })
331 } else {
332 await actualizarIp()
333 }
334}

336// Subir un archivo .gcode al servidor
337function alSubirArchivo(event) {
338 const archivo = event.files[0]
339 if (archivo) store.subirGcode(archivo)
340}
341
342// Al cambiar el archivo seleccionado en el dropdown
343function alCambiarArchivo(event) {
344 if (event.value) store.seleccionarArchivo(event.value)
345}

347// Formatea una temperatura numérica a texto con unidades
348function formatearTemperatura(temp) {
349 if (temp === undefined || temp === null) return '-- °C'
350 return `${Math.round(temp)} °C`
351}
352
353// Formatea segundos a un formato legible (ej: "2h 15m" o "05m")
354function formatearTiempo(segundos) {
355 if (!segundos || isNaN(segundos)) return '--:--'
356 const h = Math.floor(segundos / 3600)
357 const m = Math.floor((segundos % 3600) / 60)
358 const pad = (n) => n.toString().padStart(2, '0')
359 return h > 0 ? `${h}h ${pad(m)}m` : `${pad(m)}m`
360}
