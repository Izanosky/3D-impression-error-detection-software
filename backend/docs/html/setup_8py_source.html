37 # Este metodo lo utilizamos para comprobar si, con las claves introducidas, se puede establecer conexión con OctoPrint correctamente
38 def comprobar_conexion(self, retries=5, delay=3):
39 for intento in range(1, retries + 1):
40 try:
41
42 # Utilizamos uno de los endpoints proporcionados por OctoPrint para verificar la conexión, concretamente el de version
43 response = requests.get(
44 f"{self.octoprint_url}/api/version",
45 headers={"X-Api-Key": self.api_key},
46 timeout=5,
47 )
48
49 # Si nos devuelve 200, significa que la conexión es correcta
50 if response.status_code == 200:
51 return True
52
53 # Si nos devuelve 403, significa que la API key es incorrecta y nos da un error de login
54 if response.status_code == 403:
55 print("API key rechazada. Verifica que sea correcta.")
56 return False
57
58 except requests.ConnectionError:
59 print(f"Intento {intento}/{retries}: OctoPrint no disponible, reintentando en {delay}s...")
60 except Exception as e:
61 print(f"Intento {intento}/{retries}: Error al conectar con OctoPrint: {e}")
62
63 if intento < retries:
64 time.sleep(delay)
65
66 return False
