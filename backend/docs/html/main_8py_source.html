75# Tarea en segundo plano: cada segundo obtiene el estado y lo envia por WebSocket
76async def broadcast_updates():
77 global _previous_printing
78
79 # Bucle infinito que se ejecuta en segundo plano durante toda la vida de la aplicacion
80 while True:
81 # Solo consultamos a OctoPrint si hay al menos un cliente (frontend) conectado al WebSocket
82 if manager.active_connections:
83 try:
84 # Obtenemos el bucle de eventos asincrono actual de FastAPI
85 loop = asyncio.get_event_loop()
86
87 raw = await loop.run_in_executor(None, octoprint_client.obtener_estado_impresora)
88
89 # Limpiamos y formateamos los datos
90 status = _build_status(raw)
91
92 # Enviamos el estado formateado a todos los clientes web conectados
93 await manager.broadcast({
94 "type": "update",
95 "data": {
96 "status": status,
97 "detections": {"has_errors": False, "total_detections": 0, "classes": {}},
98 },
99 })
100
101 # Logica para detectar si la impresion acaba de terminar
102 # Comprueba si la palabra "print" esta en el estado actual (ej: "Printing")
103 currently_printing = "print" in status["state"].lower()
104 # Actualizamos la variable global para la siguiente iteracion del bucle
105 _previous_printing = currently_printing
106
107 except Exception as e:
108 # Capturamos cualquier error (caida de red, timeout) para que el bucle infinito no se rompa y muera
109 print(f"Error en broadcast: {e}")
110

119# Basicamente, hace que se ejecute constantemente el bucle de broadcast

132 task.cancel()

134 # Esperamos a que la tarea termine de cerrarse

166async def websocket_endpoint(websocket: WebSocket):
167 # Aceptamos la conexion entrante y la registramos en nuestro ConnectionManager
168 await manager.connect(websocket)
169 try:
170 # Bucle infinito para escuchar los comandos que envia el frontend por este socket
171 while True:
172
173 data = await websocket.receive_json()
174 # Extraemos la accion solicitada por el usuario
175 action = data.get("action")
176
177 loop = asyncio.get_event_loop()
178
179 # Obtenemos la accion que vamos a realizar en funcion del comando
180 command_map = {
181 "pausar": octoprint_client.pausar_impresion,
182 "reanudar": octoprint_client.reanudar_impresion,
183 "cancelar": octoprint_client.cancelar_impresion,
184 "iniciar": octoprint_client.iniciar_impresion,
185 }
186
187 # Obtenemos la funcion correspondiente
188 handler = command_map.get(action)
189 if handler:
190 # las ejecutamos con run_in_executor para no bloquear el Event Loop principal
191 success = await loop.run_in_executor(None, handler)
192
193 # Respondemos al cliente confirmando si el comando se ejecuto con exito
194 await websocket.send_json({
195 "type": "command_result",
196 "action": action,
197 "success": success,
198 })
199 except WebSocketDisconnect:
200 # Si el cliente cierra la pestaña del navegador o pierde conexion
201 # salta esta excepcion y lo eliminamos de la lista de conexiones activas
202 manager.disconnect(websocket)
