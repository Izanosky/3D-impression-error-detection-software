22 # Funcion auxiliar para hacer peticiones a OctoPrint, concretamente al endpoint especificado en los argumentos
23 def _request(self, method, endpoint, **kwargs):
24 url = f"{self._get_url()}{endpoint}"
25 headers = self._get_headers()
26 timeout = kwargs.pop("timeout", 10)
27
28 try:
29 return requests.request(method, url, headers=headers, timeout=timeout, **kwargs)
30 except Exception as e:
31 print(f"[OctoPrint] Error en {method} {endpoint}: {e}")
32 return None

34 # Metodo que nos devuelve el estado de la impresora
35 def obtener_estado_impresora(self):
36
37 # Consultamos los dos endpoints de OctoPrint que necesitamos:
38 printer_resp = self._request("GET", "/api/printer") # Estado fisico: temperaturas, estado de la impresora
39 job_resp = self._request("GET", "/api/job") # Estado del trabajo: archivo, progreso, tiempo restante
40
41 # Comprobamos que la respuesta no sea None (fallo de red) y que el codigo HTTP sea 200 (exito)
42 printer_ok = printer_resp and printer_resp.status_code == 200
43 job_ok = job_resp and job_resp.status_code == 200
44
45 return {
46 # Si la peticion fue correcta parseamos el JSON, si no devolvemos un diccionario vacio
47 "printer": printer_resp.json() if printer_ok else {},
48 "job": job_resp.json() if job_ok else {},
49 # Usamos printer_ok como indicador de si la impresora esta conectada y accesible

85 # Hacemos la llamada al endpoint /api/files/local de OctoPrint, pasando como argumentos:

101 except Exception as e:
